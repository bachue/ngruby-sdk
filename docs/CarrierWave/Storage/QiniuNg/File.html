def to_file
  tempfile = Tempfile.create('qiniu_ng', nil, mode: 0o600, encoding: 'ascii-8bit')
  @client.download_url(@key).download_to(tempfile.path)
  ::File.open(tempfile.path, 'r+')
ensure
  tempfile&.close
end
