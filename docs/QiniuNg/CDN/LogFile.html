class LogFile
  attr_reader :name, :size, :created_at, :url, :md5

  # @!visibility private
  def initialize(hash)
    @name = hash['name']
    @size = hash['size']
    @created_at = Time.at(hash['mtime'])
    @url = Storage::URL.new(hash['url'])
    @md5 = hash['md5']
  end
end
