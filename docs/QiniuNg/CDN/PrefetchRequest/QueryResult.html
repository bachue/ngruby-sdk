class QueryResult
  attr_reader :request_id, :url, :state, :state_detail, :created_at, :begin_at, :end_at

  # @!visibility private
  def initialize(request_id:, url:, state:, state_detail:, created_at:, begin_at:, end_at:)
    @request_id = request_id
    @url = url
    @state = state
    @state_detail = state_detail
