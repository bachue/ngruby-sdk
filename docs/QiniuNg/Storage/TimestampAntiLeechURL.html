def set(fop: nil, style: nil, filename: nil, lifetime: nil)
  @public_url.set(fop: fop, style: style, filename: filename)
  self.lifetime = lifetime unless lifetime.nil?
  generate_timestamp_anti_leech_url!
  self
end
